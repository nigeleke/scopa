1use dioxus::prelude::*;
2use dioxus_i18n::tid;
3
4use super::points_group::PointsGroup;
5
6#[component]
7pub fn PointsGroupImage(
8    hint: Option<String>,
9    group: PointsGroup,
10    disabled: bool,
11    checked: bool,
12) -> Element {
13    let src = match group {
14        PointsGroup::Scopa => asset!("/assets/images/punteggio_scopa.png"),
15        PointsGroup::CardsCount => asset!("/assets/images/punteggio_carte.png"),
16        PointsGroup::CoinsCount => asset!("/assets/images/punteggio_denari.png"),
17        PointsGroup::Settebello => asset!("/assets/images/punteggio_settebello.png"),
18        PointsGroup::Premiera => asset!("/assets/images/punteggio_premiera.png"),
19    };
20
21    rsx! {
22        document::Link { rel: "stylesheet", href: asset!("/assets/css/domain/points_group_image.css") }
23        div {
24            class: "points-group-image",
25            class: if disabled { "disabled" },
26            class: if checked { "checked" },
27            img {
28                src: src,
29                alt: tid!(&format!("{}-icon.alt-text", group)),
30            }
31            if let Some(hint) = hint {
32                span {
33                    class: "points-group-image-hint",
34                    "{hint}"
35                }
36            }
37        }
38    }
39}
