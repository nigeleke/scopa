10#[component]
11pub fn PlayingGame(game: ReadSignal<Game<Playing>>, onchange: EventHandler<State>) -> Element {
12    let round_number = game.read().history().round_number();
13    let target = game.read().target();
14
15    let undo = move |_| {
16        let playing_state = game().undo()?;
17        onchange.call(State::from(playing_state));
18        Ok(())
19    };
20
21    let mut round = use_signal(Round::default);
22
23    let score_round = move |_| {
24        let result = game().score_round(&round.read())?;
25        match result {
26            ScoreRoundResult::Playing(game) => onchange.call(State::from(game)),
27            ScoreRoundResult::Finished(game) => onchange.call(State::from(game)),
28        }
29        round.set(Round::default());
30        Ok(())
31    };
32
33    rsx! {
34        document::Link { rel: "stylesheet", href: asset!("/assets/css/pages/playing_game.css") }
35        div {
36            class: "playing-game",
37            div {
38                class: "playing-game__header",
39                RoundNumberView { value: round_number, }
40                TargetView { value: target, }
41            }
42            RoundEditor {
43                game: game(),
44                round: round,
45            }
46            div {
47                class: "playing-game__controls",
48                UndoIcon {
49                    can_undo: game().can_undo(),
50                    on_click: undo,
51                }
52                ScoreButton {
53                    can_score: round.read().is_well_defined(),
54                    onscore: score_round,
55                }
56            }
57        }
58    }
59}
