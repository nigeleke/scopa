1use crate::components::prelude::*;
2use crate::domain::prelude::*;
3
4use dioxus::prelude::{document::*, *};
5use dioxus_sdk::storage::{use_storage, LocalStorage};
6
7#[component]
8pub fn Home() -> Element {
9    let default_target = use_storage::<LocalStorage, _>("default_target".into(), Target::default);
10    let mut game =
11        use_storage::<LocalStorage, _>("game".into(), || GameState::new(default_target()));
12    provide_context(game);
13
14    let update_game = move |new_game| {
15        game.set(new_game);
16    };
17
18    rsx! {
19        Link { rel: "icon", href: asset!("/assets/favicon.ico") }
20        Link { rel: "stylesheet", href: asset!("/assets/css/main.css") }
21        Meta { name: "description", content: "Score your Scopa game using this website. This app is a free and easy to use program to help you score your Scopa card game." }
22        Meta { name: "category", content: "game" }
23        Meta { name: "keywords", content: "cards,game,scorer,scopa" }
24        Meta { name: "author", content: "Nigel Eke" }
25        header { ScopaHeader {} }
26        main {
27            match game() {
28                GameState::Starting(game) => rsx! { StartingGame { state: game.state(), onchange: update_game, } },
29                GameState::Playing(game) => rsx! { PlayingGame { state: game.state(), onchange: update_game } },
30                GameState::Finished(game) => rsx! { FinishedGame { state: game.state(), onchange: update_game } },
31            }
32        }
33        footer { ScopaFooter {} }
34    }
35}
