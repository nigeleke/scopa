1use dioxus::prelude::*;
2use dioxus_i18n::{prelude::*, unic_langid::langid};
3use dioxus_sdk::storage::{LocalStorage, use_storage};
4
5use super::{
6    consts::{STORAGE_STATE, STORAGE_TARGET},
7    i18n::{config, use_user_preferred_language},
8    routes::Route,
9    state::State,
10};
11use crate::domain::Target;
12
13const FAVICON: Asset = asset!("/assets/favicon.ico");
14const MAIN_CSS: Asset = asset!("/assets/css/main.css");
15
16#[component]
17pub fn App() -> Element {
18    let _ = use_init_i18n(|| config(langid!("en-GB")));
19    let language = use_user_preferred_language();
20    provide_context(language);
21
22    let default_target = use_storage::<LocalStorage, _>(STORAGE_TARGET.into(), Target::default);
23    let state = use_storage::<LocalStorage, _>(STORAGE_STATE.into(), || {
24        State::from(*default_target.read())
25    });
26    provide_context(state);
27
28    rsx! {
29        document::Link { rel: "icon", href: FAVICON }
30        document::Link { rel: "stylesheet", href: MAIN_CSS }
31        document::Meta { name: "description", content: "Score your Scopa game using this website. This app is a free and easy to use program to help you score your Scopa card game." }
32        document::Meta { name: "category", content: "game" }
33        document::Meta { name: "keywords", content: "cards,game,scorer,scopa" }
34        document::Meta { name: "author", content: "Nigel Eke" }
35
36        Router::<Route> {}
37    }
38}
