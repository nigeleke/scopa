1use dioxus::prelude::*;
2use dioxus_i18n::tid;
3
4use super::button::Button;
5use crate::{domain::Game, ui::state::State};
6
7#[component]
8pub fn Errors(context: ErrorContext) -> Element {
9    let mut state = use_context::<Signal<State>>();
10    let mut context = use_signal(|| context);
11
12    let onreset = move |_| {
13        let game = Game::default();
14        state.set(State::from(game));
15        context.write().clear_errors();
16    };
17
18    rsx! {
19        h2 { {tid!("error.apology")} }
20        p {
21            {tid!("error.report0")}
22            {" "}
23            Link { to: "https://github.com/nigeleke/scopa/issues", {tid!("error.report1")} }
24            {" "}
25            {tid!("error.report2")}
26        }
27        ul {
28            for (i, error) in context.read().errors().iter().enumerate() {
29                li { key: "{i}", "{error}" }
30            }
31        }
32        Button { on_click: onreset, {tid!("reset-button.text")} }
33    }
34}
