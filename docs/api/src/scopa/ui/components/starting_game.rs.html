15#[component]
16pub fn StartingGame(game: Game<Starting>, onchange: EventHandler<State>) -> Element {
17    let mut game = use_signal(|| game);
18    let mut default_target = use_storage::<LocalStorage, _>(STORAGE_TARGET.into(), Target::default);
19
20    let target = game.read().target();
21    let update_target = move |new_target| {
22        game.write().set_target(new_target);
23        default_target.set(new_target);
24    };
25
26    let teams = Teams::from(game.read().teams().clone());
27    let add_team = move |team| {
28        game.write().add_team(team);
29    };
30    let remove_team = move |id| {
31        game.write().remove_team(id)?;
32        Ok(())
33    };
34
35    let start_game = move |_| {
36        let game = game().start()?;
37        onchange.call(State::from(game));
38        Ok(())
39    };
40
41    rsx! {
42        document::Link { rel: "stylesheet", href: asset!("/assets/css/pages/starting_game.css") }
43        div {
44            class: "starting-game",
45            TargetEditor {
46                value: target,
47                onchange: update_target,
48            }
49            AddTeam {
50                onadd: add_team,
51            }
52            StartAction {
53                can_start: game.read().can_start(),
54                onstart: start_game,
55            }
56            TeamRows {
57                teams: teams,
58                onremove: remove_team,
59            }
60            HelpAction {}
61        }
62    }
63}

65#[component]
66fn AddTeam(onadd: EventHandler<Team>) -> Element {
67    let mut team_name = use_signal(TeamName::default);
68
69    let mut add_team = move |value: TeamName| {
70        let team = Team::from(value.to_string().as_str());
71        onadd.call(team);
72        team_name.set(TeamName::default());
73    };
74
75    let add_team_button_action = move |_| add_team(team_name());
76
77    rsx! {
78        span {
79            TeamNameEditor {
80                team_name: team_name,
81                autofocus: true,
82                oncommit: add_team,
83                placeholder: tid!("team-name-editor.placeholder"),
84            }
85            " "
86            Button { on_click: add_team_button_action, " + " }
87        }
88    }
89}

91#[component]
92fn TeamRows(teams: Vec<Team>, onremove: EventHandler<TeamId>) -> Element {
93    let remove_team = move |team: &Team| {
94        let id = team.id();
95        move |_event| {
96            onremove.call(id);
97        }
98    };
99
100    let team_row = move |team: &Team| {
101        rsx! {
102            tr {
103                td { Button { on_click: remove_team(team),  "-"  } }
104                td { TeamNameView { value: team.name().clone() } }
105            }
106        }
107    };
108
109    rsx! {
110        table {
111            hidden: teams.is_empty(),
112            { teams.iter().map(team_row) }
113        }
114    }
115}

117#[component]
118fn StartAction(can_start: bool, onstart: EventHandler<()>) -> Element {
