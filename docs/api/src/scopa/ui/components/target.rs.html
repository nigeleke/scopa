9#[component]
10pub fn TargetView(value: Target) -> Element {
11    rsx! {
12        document::Link { rel: "stylesheet", href: asset!("/assets/css/domain/target.css") }
13        Container {
14            {tid!("points-view.text", n: value.to_string())}

19#[component]
20pub fn TargetEditor(value: Target, onchange: EventHandler<Target>) -> Element {
21    let update_target = move |value: String| {
22        let result = Target::from_str(value.as_str());
23        if let Ok(new_value) = result {
24            onchange.call(new_value);
25        };
26    };
27
28    rsx! {
29        document::Link { rel: "stylesheet", href: asset!("/assets/css/domain/target.css") }
30        Container {
31            {tid!("points-editor.prefix")}
32            label {
33                Input {
34                    typ: "number",
35                    value: "{value}",
36                    on_input: update_target,
37                    min: "11",
38                    step: "5",
39                    size: "3",
40                    aria_label: tid!("points-editor.aria-label"),
41                }
42            }
43            {tid!("points-editor.suffix")}
44        }
45    }
46}
