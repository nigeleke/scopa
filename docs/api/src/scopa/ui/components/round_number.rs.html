1use dioxus::prelude::*;
2use dioxus_i18n::tid;
3use gloo_timers::future::sleep;
4
5use crate::domain::*;
6
7#[component]
8pub fn RoundNumberView(value: ReadOnlySignal<RoundNumber>) -> Element {
9    let mut pulse = use_signal(|| false);
10
11    use_effect(move || {
12        let _ = value();
13        pulse.set(true);
14        spawn(async move {
15            sleep(std::time::Duration::from_millis(1000)).await;
16            pulse.set(false);
17        });
18    });
19
20    rsx! {
21        document::Link { rel: "stylesheet", href: asset!("/assets/css/domain/round_number.css") }
22        Container {
23            pulse: pulse(),
24            {tid!("round-view.text")}
25            { " " }
26            { value.to_string() }
27        }
28    }
29}
30
31#[component]
32fn Container(pulse: bool, children: Element) -> Element {
33    rsx! {
34        document::Link { rel: "stylesheet", href: asset!("/assets/css/domain/round_number.css") }
35        div {
36            class: "round-number-outer",
37            class: if pulse { "round-number-pulse" },
38            div {
39                class: "round-number-inner",
40                {children},
41            },
42        }
43    }
44}
