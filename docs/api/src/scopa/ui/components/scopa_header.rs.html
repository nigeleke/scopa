20#[component]
21pub fn ScopaHeader() -> Element {
22    rsx! {
23        document::Link { rel: "stylesheet", href: asset!("/assets/css/ui/scopa_header.css") }
24        div {
25            class: "scopa_header",
26            Menu {}
27            Glow { {tid!("scopa-app.title-text")} }
28        }
29    }
30}

32#[component]
33fn Menu() -> Element {
34    rsx! {
35        nav {
36            MenuIcon { popovertarget: MAIN_MENU_ID }
37            div {
38                class: "scopa_header_popover_menu",
39                id: MAIN_MENU_ID,
40                popover: "auto",
41                Flag { src: asset!("/assets/images/flags/gb.svg"), lang: "en-GB" }
42                Flag { src: asset!("/assets/images/flags/it.svg"), lang: "it-IT" }
43                RestartMenuItem {}
44            },
45        }
46    }
47}

49#[component]
50fn Flag(src: String, lang: String) -> Element {
51    let alt = tid!(&format!("lang.{}", lang));
52    let lang = Language::from_str(lang.as_str()).ok();
53
54    let mut i18n = use_context::<Signal<Option<Language>>>();
55
56    let on_click = move |_| {
57        i18n.set(lang.clone());
58    };
59
60    rsx! {
61        button {
62            onclick: on_click,
63            img {
64                class: "flag",
65                src,
66                alt,
67            }
68        }
69    }
70}
71
72#[component]
73fn RestartMenuItem() -> Element {
74    let default_target = use_storage::<LocalStorage, _>(STORAGE_TARGET.into(), Target::default);
75    let mut state = use_context::<Signal<State>>();
76
77    let restart_game = move || state.set(State::from(Game::from(default_target())));
78
79    match state() {
80        State::Starting(_) => rsx! {},
81        _ => rsx! { RestartIcon { on_click: restart_game } },
82    }
83}
