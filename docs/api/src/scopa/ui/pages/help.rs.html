1use dioxus::prelude::*;
2use dioxus_i18n::{tid, unic_langid::langid};
3
4use crate::ui::{components::HomeIcon, i18n::Language, routes::Route};
5
6#[component]
7pub fn Help() -> Element {
8    let i18n = use_context::<Signal<Option<Language>>>();
9
10    let mut region = use_signal(String::default);
11    let mut starting_image = use_signal(String::default);
12    let mut scoring_image = use_signal(String::default);
13
14    use_effect(move || {
15        region.set(
16            i18n()
17                .map_or(langid!("en-GB"), |l| l.identifier())
18                .language
19                .to_string(),
20        );
21        match region().as_str() {
22            "it" => {
23                starting_image.set(asset!("assets/images/starting-it.png").to_string());
24                scoring_image.set(asset!("assets/images/scoring-it.png").to_string())
25            }
26            _ => {
27                starting_image.set(asset!("assets/images/starting-en.png").to_string());
28                scoring_image.set(asset!("assets/images/scoring-en.png").to_string())
29            }
30        }
31    });
32
33    let navigator = use_navigator();
34    let on_home = move |_| {
35        navigator.push(Route::Home);
36    };
37
38    rsx! {
39        div {
40            class: "help",
41            document::Link { rel: "stylesheet", href: asset!("/assets/css/pages/help.css") }
42            h1 { {tid!("help.heading")} }
43            p { {tid!("help.intro")} }
44            div {
45                h2 { {tid!("help.starting-heading")} }
46                p { {tid!("help.starting-intro")} }
47                img { src: "{starting_image}" }
48                p { {tid!("help.starting-points")} }
49                p { {tid!("help.starting-add-team")} }
50                p { {tid!("help.starting-remove-team")} }
51                p { {tid!("help.starting-start-game")} }
52                h2 { {tid!("help.scoring-heading")} }
53                p { {tid!("help.scoring-intro")} }
54                img { src: "{scoring_image}" }
55                p { {tid!("help.scoring-scopa")} }
56                p { {tid!("help.scoring-basics")} }
57                p { {tid!("help.scoring-undo")} }
58            }
59            HomeIcon { on_click: on_home, }
60        }
61    }
62}
