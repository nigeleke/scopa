1use std::str::FromStr;
2
3use dioxus::prelude::*;
4use dioxus_i18n::prelude::*;
5use dioxus_sdk::storage::*;
6
7use super::language::Language;
8use crate::ui::consts::STORAGE_LANGUAGE;
9
10pub fn use_user_preferred_language() -> Signal<Option<Language>> {
11    const FALLBACK_LANGUAGE: &str = "EN-GB";
12
13    let document_language = use_resource(|| async {
14        let mut eval = document::eval("dioxus.send(navigator.language)");
15        let lang = eval
16            .recv::<String>()
17            .await
18            .unwrap_or_else(|_| String::from(FALLBACK_LANGUAGE));
19        Language::from_str(lang.as_str())
20            .unwrap_or_else(|_| Language::from_str(FALLBACK_LANGUAGE).expect("valid fallback"))
21    });
22
23    #[allow(clippy::redundant_closure)]
24    let mut preferred_language =
25        use_storage::<LocalStorage, _>(STORAGE_LANGUAGE.into(), || document_language());
26    let preferred_language_is_defined = use_signal(|| preferred_language.read().is_some());
27
28    use_effect(move || {
29        if !*preferred_language_is_defined.read() {
30            preferred_language.set(document_language());
31        }
32    });
33
34    let mut i18n = use_context::<I18n>();
35
36    use_effect(move || {
37        if let Some(lang) = preferred_language.read().as_ref() {
38            i18n.set_language(lang.identifier());
39        }
40    });
41
42    preferred_language
43}
