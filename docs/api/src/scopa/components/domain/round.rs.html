7#[component]
8pub fn RoundEditor(state: PlayingState, round: Signal<Round>) -> Element {
9    let teams = Vec::from(state.teams());
10
11    let first_active_team = teams.iter().find(|t| t.is_playing()).unwrap();
12    let first_active_team_id = first_active_team.id();
13
14    let leading_team_score = teams
15        .iter()
16        .map(|team| state.points(team.id()))
17        .max()
18        .unwrap_or_default();
19
20    let none_column_components = [
21        rsx! { Empty {} },
22        rsx! { ScopaIcon { hint: tid!("scopa-icon.hint") } },
23        rsx! { RadioTeamIcon { hint: tid!("cards-count-icon.hint"), group: PointsGroup::CardsCount, team: None, round: round } },
24        rsx! { RadioTeamIcon { hint: tid!("coins-count-icon.hint"), group: PointsGroup::CoinsCount, team: None, round: round } },
25        rsx! { RadioTeamIcon { hint: tid!("settebello-icon.hint"), group: PointsGroup::Settebello, team: None, round: round } },
26        rsx! { RadioTeamIcon { hint: tid!("premiera-icon.hint"), group: PointsGroup::Premiera, team: None, round: round } },
27    ];
28
29    let team_column_components = move |team: &Team| {
30        let id = team.id();
31        let name = team.name();
32        let points = state.points(id);
33        let is_leader = points == leading_team_score;
34        let is_not_playing = team.is_not_playing();
35        [
36            rsx! { TeamHeader { name: name, points: points, is_leader: is_leader } },
37            rsx! { ScopaScore { team: team.clone(), round: round, autofocus: id == first_active_team_id, disabled: is_not_playing } },
38            rsx! { RadioTeamIcon { group: PointsGroup::CardsCount, team: Some(team.clone()), round: round, disabled: is_not_playing } },
39            rsx! { RadioTeamIcon { group: PointsGroup::CoinsCount, team: Some(team.clone()), round: round, disabled: is_not_playing } },
40            rsx! { RadioTeamIcon { group: PointsGroup::Settebello, team: Some(team.clone()), round: round, disabled: is_not_playing } },
41            rsx! { RadioTeamIcon { group: PointsGroup::Premiera, team: Some(team.clone()), round: round, disabled: is_not_playing } },
42        ]
43    };
44
45    let some_column_components = teams.iter().map(team_column_components).collect::<Vec<_>>();
46    let columns_count = some_column_components.len() + 1;
47
48    rsx! {
49        Link { rel: "stylesheet", href: asset!("/assets/css/domain/round.css") }
50        div {
51            class: "round-editor-container",
52            for j in 0..columns_count {
53                for i in 0..6 {
54                    if j == 0 {
55                        {none_column_components[i].clone()}
56                    } else {
57                        {some_column_components[j-1][i].clone()}
58                    }
59                }
60            }
61        }
62    }
63}

106#[component]
107fn ScopaScore(team: Team, round: Signal<Round>, autofocus: bool, disabled: bool) -> Element {
108    let id = team.id();
109    let name = &team.name();
110
111    let mut draft = use_signal(Points::default);
112
113    use_effect(move || draft.set(round.read().scopas(id)));
114
115    let update_draft = move |points| {
116        round.set(round().with_scopas(id, points));
117    };
118
119    rsx! {
120        PointsEditor {
121            value: draft(),
122            onchange: update_draft,
123            autofocus: autofocus,
124            disabled: disabled,
125            aria_label: tid!("score-scopa-editor.aria-label", teamname: name.to_string()),
126        }
127    }
128}

130#[component]
131fn RadioTeamIcon(
132    hint: Option<String>,
133    group: PointsGroup,
134    team: Option<Team>,
135    round: Signal<Round>,
136    #[props(default = false)] disabled: bool,
137) -> Element {
138    let id = team.clone().map(|t| t.id());
139    let name = team.map(|t| t.name());
140
141    let mut draft = use_signal(move || None);
142
143    use_effect(move || {
144        let selection = match group {
145            PointsGroup::Scopa => unreachable!(),
146            PointsGroup::CardsCount => round.read().card_count(),
147            PointsGroup::CoinsCount => round.read().coins_count(),
148            PointsGroup::Settebello => round.read().settebello(),
149            PointsGroup::Premiera => round.read().premiera(),
150        };
151
152        draft.set(selection);
153    });
154
155    let update_draft = move |_| {
156        let new_round = match group {
157            PointsGroup::Scopa => unreachable!(),
158            PointsGroup::CardsCount => round().with_highest_card_count(id),
159            PointsGroup::CoinsCount => round().with_highest_coins_count(id),
160            PointsGroup::Settebello => id.map_or(round(), |id| round().with_settobello(id)),
161            PointsGroup::Premiera => round().with_premiera(id),
162        };
163
164        round.set(new_round);
165    };
166
167    let is_checked = *draft.read() == id;
168    let settebello_none_disabled =
169        group == PointsGroup::Settebello && id.is_none() && draft.read().is_some();
170    let is_disabled = disabled || settebello_none_disabled;
171
172    rsx! {
173        label {
174            class: "radio-team-icon",
175            Input {
176                typ: "radio",
177                name: group.to_string(),
178                value: format!("{}-{}", group.to_string(), id.map_or("none".to_string(), |t| t.to_string())),
179                on_input: update_draft,
180                aria_label: tid!("score-group-icon.aria-label", group: group.to_string(), teamname: name.map_or("no one".to_string(), |n| n.to_string())),
181                checked: is_checked,
182                disabled: is_disabled,
183            }
184            PointsGroupImage {
185                hint,
186                group: group,
187                disabled: is_disabled,
188                checked: is_checked,
189            }
190        }
191    }
