8#[component]
9pub fn FinishedGame(state: FinishedState, onchange: EventHandler<GameState>) -> Element {
10    let default_target = use_storage::<LocalStorage, _>("default_target".into(), Target::default);
11
12    let winner = state.winner();
13
14    let mut teams = Vec::from(state.teams());
15    teams.sort_by_key(|b| std::cmp::Reverse(state.points(b.id())));
16
17    let team_points = teams
18        .into_iter()
19        .map(|team: Team| rsx! { TeamPoints { state: state.clone(), team: team } })
20        .collect::<Vec<_>>();
21
22    let mut retain_settings = use_signal(|| true);
23
24    let update_retain_settings = move |value: String| {
25        retain_settings.set(str::parse::<bool>(&value).unwrap());
26    };
27
28    let start_new_game = move || {
29        let mut new_game = Game::new_starting_state(default_target());
30
31        if retain_settings() {
32            let teams = Vec::from(state.teams());
33            teams.iter().for_each(|t| {
34                let team = Team::new(&t.name());
35                let _ = new_game.add_team(team);
36            });
37        }
38
39        if retain_settings() {
40            onchange.call(new_game.start().unwrap());
41        } else {
42            onchange.call(GameState::Starting(new_game));
43        }
44    };
45
46    rsx! {
47        Link { rel: "stylesheet", href: asset!("/assets/css/state/finished_game.css") }
48        div {
49            class: "finished-game-container",
50            div {
51                class: "finished-game-winner-text",
52                Glow { {tid!("winner-view.text", teamname: winner.to_string())} }
53            }
54            div {
55                class: "finished-game-team-scores",
56                for team_points in team_points {
57                    {team_points}
58                }
59            }
60            div {
61                class: "finished-game-controls",
62                div {
63                    class: "finished-game-start-game-button",
64                    Button {
65                        on_click: start_new_game,
66                        {tid!("start-new-game-button.text")}
67                    }
68                }
69                div {
70                    class: "finished-game-retain-group",
71                    div {
72                        class: "finished-game-retain-checkbox",
73                        Input {
74                            typ: "checkbox",
75                            value: retain_settings(),
76                            checked: retain_settings(),
77                            on_input: update_retain_settings,
78                            aria_label: tid!("start-new-game-settings.aria-label"),
79                        }
80                        { tid!("start-new-game-settings.text") }
81                    }
82                }
83            }
84        }
85    }
86}
