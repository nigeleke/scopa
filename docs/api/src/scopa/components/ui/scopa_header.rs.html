7use dioxus::prelude::{document::*, *};

11#[component]
12pub fn ScopaHeader() -> Element {
13    rsx! {
14        Link { rel: "stylesheet", href: asset!("/assets/css/ui/scopa_header.css") }
15        div {
16            class: "scopa_header",
17            Menu {}
18            Glow { {tid!("scopa-app.title-text")} }
19        }
20    }
21}
22
23#[component]
24fn Menu() -> Element {
25    const MAIN_MENU_ID: &str = "main-menu";
26    rsx! {
27        nav {
28            MenuIcon { popovertarget: MAIN_MENU_ID }
29            div {
30                class: "scopa_header_popover_menu",
31                id: MAIN_MENU_ID,
32                popover: "auto",
33                Flag { src: asset!("/assets/images/flags/gb.svg"), lang: "en-GB" }
34                Flag { src: asset!("/assets/images/flags/it.svg"), lang: "it-IT" }
35                RestartMenuItem {}
36            },
37        }
38    }
39}
40
41#[component]
42fn Flag(src: String, lang: String) -> Element {
43    let alt = tid!(&format!("lang.{}", lang));
44    let lang = Language::try_from(lang).ok();
45
46    let mut i18n = use_context::<Signal<Option<Language>>>();
47
48    let on_click = move |_| {
49        i18n.set(lang.clone());
50    };
51
52    rsx! {
53        button {
54            onclick: on_click,
55            img {
56                class: "flag",
57                src,
58                alt,
59            }
60        }
61    }
62}
63
64#[component]
65fn RestartMenuItem() -> Element {
66    let default_target = use_storage::<LocalStorage, _>("default_target".into(), Target::default);
67    let mut state = use_context::<Signal<GameState>>();
68
69    let restart_game = move || state.set(GameState::new(default_target()));
70
71    match state() {
72        GameState::Starting(_) => rsx! {},
73        _ => rsx! { RestartIcon { on_click: restart_game } },
74    }
75}
