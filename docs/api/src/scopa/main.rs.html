18fn main() {

23fn app() -> Element {
24    let document_language = use_resource(move || async move {
25        let mut eval = document::eval("dioxus.send(navigator.language)");
26        let lang = Language::try_from(eval.recv::<String>().await.unwrap()).unwrap();
27        info!("use_resource::lang: {:?}", lang);
28        lang
29    });
30
31    #[allow(clippy::redundant_closure)]
32    let mut preferred_language =
33        use_storage::<LocalStorage, _>("lang".into(), || document_language());
34    let preferred_language_is_defined = use_signal(|| (*preferred_language.read()).is_some());
35
36    use_effect(move || {
37        if !*preferred_language_is_defined.read() {
38            preferred_language.set(document_language());
39        }
40    });
41
42    provide_context(preferred_language);
43
44    let mut i18n = use_init_i18n(|| i18n::config(langid!("en-GB")));
45    use_effect(move || {
46        if let Some(preferred_language) = preferred_language.read().as_ref() {
47            dioxus::logger::tracing::info!("Setting preferred language {:?}", preferred_language);
48            i18n.set_language(preferred_language.identifier());
49        }
50    });
51
52    rsx! { Home {} }
53}
